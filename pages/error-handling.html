Error Handling

Exceptions

  1. Likt som i mange andre programmeringsspråk, men brukes sjelden
Venn *venn = [[Venn alloc] init];

@try 
{
    [venn sladre];
}
@catch (NSException *exception) 
{
    NSLog(@"Feilet %@: %@", [exception name], [exception reason]);
}

	// Noe feilet
}

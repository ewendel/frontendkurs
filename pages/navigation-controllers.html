Storyboard: Sende data mellom ViewControllers i et NavigationView

// I "MasterViewController"
- (void)prepareForSegue:(UIStoryboardSegue *)segue sender:(id)sender
{
	// Få tak i destinasjonen og sett data på den før vi navigerer.
	// Krever at "track" er en property på destinasjonskontrolleren
    SpotifyPlayerViewController *viewController = (SpotifyPlayerViewController *)[segue destinationViewController];
    viewController.track = self.tracks[[[self.tableView indexPathForSelectedRow] row]];
}
